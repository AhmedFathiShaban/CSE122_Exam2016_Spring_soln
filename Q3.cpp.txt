#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

int main()
{
    // This will give different numbers in each new run
    srand(time(NULL));

    double numbers[10000];

    int frequency[10];

    for(int i = 0 ; i<10000 ; i++)
    {
        // rand() generates an integer between 0 and RAND_MAX , which is a large constant value known by the compiler
        numbers[i] = (1.0*rand())/RAND_MAX ;
        // OR You could use ::   numbers[i] = (1.0*rand()%1000000)/1000000 ;

        // update the frequency array
            if(numbers[i] <= .1)
            frequency[0]++;
        else if(numbers[i] <= .2)
            frequency[1]++;
        else if(numbers[i] <= .3)
            frequency[2]++;
        else if(numbers[i] <= .4)
            frequency[3]++;
        else if(numbers[i] <= .5)
            frequency[4]++;
        else if(numbers[i] <= .6)
            frequency[5]++;
        else if(numbers[i] <= .7)
            frequency[6]++;
        else if(numbers[i] <= .8)
            frequency[7]++;
        else if(numbers[i] <= .9)
            frequency[8]++;
        else
            frequency[9]++;
    }

    double lowerLimit = 0;
    double upperLimit = .1;

        for(int i = 0 ; i<10 ; i++)
    {
        cout << " The frequency of numbers in range from " << lowerLimit << " : " << upperLimit << " is " << frequency[i] <<endl;
             lowerLimit += .1;
             upperLimit  = lowerLimit + .1 ;
    }

        int MaxFrequency = -1 ;
        int maxIndex = -1 ;

        for(int i = 0 ; i<10 ; i++)
        {
            if(frequency[i] > MaxFrequency)
            {
                MaxFrequency = frequency[i];
                maxIndex = i;
            }
        }

        cout << "The band that has the highest frequency is band " << maxIndex << endl;
        cout << "With values ranging from " << maxIndex*.1 << " to " << (maxIndex+1)*.1 <<endl;
        cout << "With Frequency of " << MaxFrequency <<endl;

}
